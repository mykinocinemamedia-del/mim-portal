   272→  /**
   273→   * Phase 2: Send signing reminder WhatsApp to unsigned contracts older than 3 days.
   274→   * Uses notes field marker '[REMINDER_SENT]' to avoid duplicates.
   275→   */
   276→  private async sendSigningReminders(stats: any): Promise<number> {
   277→    let sent = 0
   278→
   279→    try {
   280→      const now = Date.now()
   281→      const reminderThreshold = new Date(now - REMINDER_AFTER_DAYS * DAY_MS)
   282→
   283→      // Contracts that are draft, not fully signed, older than 3 days, no reminder yet
   284→      const unsignedContracts = await db.contract.findMany({
   285→        where: {
   286→          status: 'draft',
   287→          createdAt: { lte: reminderThreshold },
   288→          // Hasn't been reminded yet
   289→          content: { not: { contains: '[CONTRACT_REMINDER_SENT]' } },
   290→          // Not fully signed
   291→          OR: [
   292→            { signedHelper: false },
   293→            { signedEmployer: false },
   294→            { signedAdmin: false },
   295→          ],
   296→        },
   297→        include: {
   298→          helper: { select: { id: true, fullName: true, phone: true } },
   299→          employer: { select: { id: true, fullName: true, phone: true } },
   300→          booking: { select: { id: true } },
   301→        },
   302→        take: 30,
   303→      })
   304→
   305→      // Group by booking to avoid sending duplicate reminders for the same booking
   306→      const byBooking = new Map<string, typeof unsignedContracts>()
   307→      for (const c of unsignedContracts) {
   308→        const key = c.bookingId || c.id
   309→        if (!byBooking.has(key)) byBooking.set(key, [])
   310→        byBooking.get(key)!.push(c)
   311→      }
   312→
   313→      for (const [, contracts] of byBooking) {
   314→        try {
   315→          const first = contracts[0]
   316→          const helper = first.helper
   317→          const employer = first.employer
   318→
   319→          const contractLink = `${process.env.NEXT_PUBLIC_SITE_URL || 'https://mim-portal.vercel.app'}/admin/contracts`
   320→
   321→          // Build reminder message
   322→          const buildReminder = (name: string, role: 'helper' | 'employer') => {
   323→            const roleLabel = role === 'helper' ? 'Pembantu' : 'Majikan'
   324→            return (
   325→              `📋 Peringatan Tandatangan Kontrak MIM Portal\n\n` +
   326→              `Hai ${name},\n\n` +
   327→              `Kontrak anda sebagai ${roleLabel} masih belum ditandatangani. ` +
   328→              `Sila tandatangani sebelum kontrak tamat tempoh (7 hari dari tarikh dijana).\n\n` +
   329→              `🔗 Link kontrak: ${contractLink}\n\n` +
   330→              `Jika ada soalan, hubungi admin:\n` +
   331→              `WhatsApp: ${COMPANY.phone}\n` +
   332→              `Email: ${COMPANY.email}\n\n` +
   333→              `Terima kasih! 🙏`
   334→            )
   335→          }
   336→
   337→          let thisSent = 0
   338→          if (helper?.phone) {
   339→            const r = await sendWhatsApp({
   340→              to: helper.phone,
   341→              body: buildReminder(helper.fullName, 'helper'),
   342→            })
   343→            if (r.success) {
   344→              thisSent++
   345→              stats.whatsappSent++
   346→            }
   347→          }
   348→          if (employer?.phone) {
   349→            const r = await sendWhatsApp({
   350→              to: employer.phone,
   351→              body: buildReminder(employer.fullName, 'employer'),
   352→            })
   353→            if (r.success) {
   354→              thisSent++
   355→              stats.whatsappSent++
   356→            }
   357→          }
   358→
   359→          // Mark reminder as sent on all 3 contracts in this booking
   360→          for (const c of contracts) {
   361→            await db.contract.update({
   362→              where: { id: c.id },
   363→              data: {
   364→                content: `${c.content || ''}\n\n[CONTRACT_REMINDER_SENT ${new Date().toISOString()}]`,
   365→              },
   366→            })
   367→          }
   368→
   369→          sent++
   370→
   371→          await this.logActivity({
   372→            action: 'send_contract_reminder',
   373→            status: 'success',
   374→            input: JSON.stringify({ bookingId: first.bookingId }),
   375→            output: JSON.stringify({ whatsappSent: thisSent }),
   376→          })
   377→        } catch (e: any) {
   378→          stats.errors++
   379→          await this.logActivity({
   380→            action: 'send_contract_reminder',
   381→            status: 'error',
   382→            input: JSON.stringify({ bookingId: contracts[0]?.bookingId }),
   383→            errorMessage: e.message || String(e),
   384→          })
   385→        }
   386→      }
   387→    } catch (e: any) {
   388→      stats.errors++
   389→    }
   390→
   391→    return sent
   392→  }

   400→    try {

   405→      const staleContracts = await db.contract.findMany({
   406→        where: {
   407→          status: 'draft',
   408→          createdAt: { lte: expireThreshold },
   409→          OR: [
   410→            { signedHelper: false },
   411→            { signedEmployer: false },
   412→            { signedAdmin: false },
   413→          ],
   414→        },
   415→        include: {
   416→          helper: { select: { id: true, fullName: true, phone: true } },
   417→          employer: { select: { id: true, fullName: true, phone: true } },
   418→          booking: { select: { id: true } },

   420→        take: 30,
   421→      })
   422→
   423→      for (const contract of staleContracts) {
   424→        try {
   425→          await db.contract.update({
   426→            where: { id: contract.id },
   427→            data: { status: 'expired' },

   440→            action: 'expire_contract',

   469→      }
   470→    } catch (e: any) {
   471→      stats.errors++
   472→    }
   473→
   474→    return expired
   475→  }

   477→  /**
   478→   * Phase 4: Activate contracts that are fully signed by all 3 parties.
   479→   * When a booking's all 3 contracts are active, update booking status to 'active'.
   480→   */
   481→  private async activateFullySignedContracts(stats: any): Promise<number> {
   482→    let activated = 0
   483→
   484→    try {
   485→      // Find draft contracts fully signed by all 3 parties
   486→      const readyContracts = await db.contract.findMany({
   487→        where: {
   488→          status: 'draft',
   489→          signedHelper: true,
   490→          signedEmployer: true,
   491→          signedAdmin: true,
   492→        },
   493→        include: {
   494→          booking: { select: { id: true, status: true } },
   495→          helper: { select: { id: true, fullName: true, phone: true } },
   496→          employer: { select: { id: true, fullName: true, phone: true } },
   497→        },
   498→        take: 30,
   499→      })
   500→
   501→      const bookingIdsToCheck = new Set<string>()
   502→
   503→      for (const contract of readyContracts) {
   504→        try {
   505→          await db.contract.update({
   506→            where: { id: contract.id },
   507→            data: { status: 'active' },
   508→          })
   509→          activated++
   510→
   511→          if (contract.bookingId) {
   512→            bookingIdsToCheck.add(contract.bookingId)
   513→          }
   514→
   515→          await this.logActivity({
   516→            action: 'activate_contract',
   517→            status: 'success',
   518→            input: JSON.stringify({ contractId: contract.id, type: contract.contractType }),
   519→            output: JSON.stringify({ status: 'active' }),
   520→          })
   521→        } catch (e: any) {
   522→          stats.errors++
   523→          await this.logActivity({
   524→            action: 'activate_contract',
   525→            status: 'error',
   526→            input: JSON.stringify({ contractId: contract.id }),
   527→            errorMessage: e.message || String(e),
   528→          })
   529→        }
   530→      }
   531→
   532→      // For each affected booking, check if all 3 contracts are now active
   533→      for (const bookingId of bookingIdsToCheck) {
   534→        try {
   535→          const allContracts = await db.contract.findMany({
   536→            where: { bookingId },
   537→            select: { id: true, status: true, contractType: true },
   538→          })
   539→
   540→          const allActive =
   541→            allContracts.length >= 3 &&
   542→            allContracts.every((c) => c.status === 'active')
   543→
   544→          if (allActive) {
   545→            // Update booking status to 'active' (work has started)
   546→            await db.booking.update({
   547→              where: { id: bookingId },
   548→              data: { status: 'active' },
   549→            })
   550→
   551→            await this.notify({
   552→              category: 'contract',
   553→              severity: 'info',
   554→              title: 'Semua kontrak diaktifkan!',
   555→              message: `Booking ${bookingId}: Semua 3 kontrak telah ditandatangani dan diaktifkan. Status booking dikemaskini ke 'active'.`,
   556→              actionUrl: `/admin/contracts?booking=${bookingId}`,
   557→            })
   558→
   559→            await this.logActivity({
   560→              action: 'activate_booking',
   561→              status: 'success',
   562→              input: JSON.stringify({ bookingId }),
   563→              output: JSON.stringify({ allContractsActive: true }),
   564→            })
   565→          }
   566→        } catch (e: any) {
   567→          stats.errors++
   568→        }
   569→      }
   570→    } catch (e: any) {
   571→      stats.errors++
   572→    }
   573→
   574→    return activated
   575→  }

   600→- Perkhidmatan: ${helper?.serviceType || helper?.desiredJob || 'Pembantu Rumah'}
   601→- Kemahiran: ${[helper?.skills, helper?.otherSkills].filter(Boolean).join(', ') || 'umum'}
   602→- Pengalaman: ${helper?.experience || 'tidak diketahui'}
   603→- Agama: ${helper?.religion || 'tidak diketahui'}
   604→- Lokasi: ${[helper?.city, helper?.state].filter(Boolean).join(', ') || 'Malaysia'}
   605→
   606→MAKLUMAT MAJIKAN:
   607→- Nama: ${employer?.fullName || 'tidak diketahui'}
   608→- Perkhidmatan diperlukan: ${employer?.serviceType || booking?.serviceType || 'Pembantu Rumah'}
   609→- Bilangan anak: ${employer?.numKids || 0}${employer?.kidsAges ? ` (umur: ${employer.kidsAges})` : ''}
   610→- Gaji ditawarkan: RM${employer?.salaryOffered || booking?.salary || 'tidak diketahui'}
   611→- Kriteria khusus: ${employer?.criteria || 'tiada'}
   612→- Lokasi: ${[employer?.city, employer?.state].filter(Boolean).join(', ') || 'Malaysia'}
   613→
   614→MAKLUMAT BOOKING:
   615→- Mod: ${booking?.liveIn ? 'Live-in' : 'Back & Forth'}
   616→- Tempoh: ${booking?.durationMonths || 12} bulan
   617→- Permintaan khas: ${booking?.specialRequests || 'tiada'}
   618→
   619→Cadangkan klausa khas yang sesuai untuk kontrak ini (contoh: jaga bayi, masak, larangan, keperluan khusus).`
   620→
   621→      const messages: ChatMessage[] = [
   622→        { role: 'system', content: systemPrompt },
   623→        { role: 'user', content: userPrompt },
   624→      ]
   625→      const aiRes = await aiChat(messages, { temperature: 0.4, maxTokens: 800 })
   626→      const parsed = parseAIJson<AIPersonalization>(aiRes.content)
   627→      if (!parsed) return defaultResult
   628→
   629→      return {
   630→        specialClauses: Array.isArray(parsed.specialClauses)
   631→          ? parsed.specialClauses.slice(0, 4)
   632→          : [],
   633→        responsibilities: Array.isArray(parsed.responsibilities)
   634→          ? parsed.responsibilities.slice(0, 4)
   635→          : [],
   636→        notes: typeof parsed.notes === 'string' ? parsed.notes : '',
   637→      }
   638→    } catch {
   639→      return defaultResult
   640→    }
   641→  }

   847→  /**
   848→   * Send WhatsApp to helper, employer, and admin when contracts are created.
   849→   */
   850→  private async notifyContractsCreated(booking: any): Promise<number> {
   851→    let sent = 0
   852→    const contractLink = `${process.env.NEXT_PUBLIC_SITE_URL || 'https://mim-portal.vercel.app'}/admin/contracts`
   853→
   854→    const buildMsg = (name: string, role: 'helper' | 'employer' | 'admin') => {
   855→      const roleLabel =
   856→        role === 'helper' ? 'Pembantu' : role === 'employer' ? 'Majikan' : 'Admin'
   857→      const otherParty =
   858→        role === 'helper'
   859→          ? `Majikan: ${booking.employer.fullName}`
   860→          : role === 'employer'
   861→          ? `Pembantu: ${booking.helper.fullName}`
   862→          : `Pembantu: ${booking.helper.fullName}, Majikan: ${booking.employer.fullName}`
   863→      return (
   864→        `📋 Kontrak Baru MIM Portal\n\n` +
   865→        `Hai ${name} (${roleLabel}),\n\n` +
   866→        `3 kontrak telah dijana untuk penempatan ini:\n` +
   867→        `• Agensi ↔ Pembantu\n` +
   868→        `• Agensi ↔ Majikan\n` +
   869→        `• Majikan ↔ Pembantu\n\n` +
   870→        `Butiran:\n` +
   871→        `${otherParty}\n` +
   872→        `Gaji: RM${(booking.salary || booking.employer.salaryOffered || 1800).toLocaleString('ms-MY')}/bulan\n` +
   873→        `Mod: ${booking.liveIn ? 'Live-in' : 'Back & Forth'}\n` +
   874→        `Tempoh: ${booking.durationMonths || 12} bulan\n\n` +
   875→        `Sila semak & tandatangani kontrak anda:\n` +
   876→        `🔗 ${contractLink}\n\n` +
   877→        `Kontrak perlu ditandatangani dalam 7 hari. Jika ada soalan, hubungi ${COMPANY.phone}.`
   878→      )
   879→    }
   880→
   881→    // Helper
   882→    if (booking.helper?.phone) {
   883→      try {
   884→        const r = await sendWhatsApp({
   885→          to: booking.helper.phone,
   886→          body: buildMsg(booking.helper.fullName, 'helper'),
   887→        })
   888→        if (r.success) sent++
   889→      } catch {}
   890→    }
   891→
   892→    // Employer
   893→    if (booking.employer?.phone) {
   894→      try {
   895→        const r = await sendWhatsApp({
   896→          to: booking.employer.phone,
   897→          body: buildMsg(booking.employer.fullName, 'employer'),
   898→        })
   899→        if (r.success) sent++
   900→      } catch {}
   901→    }
   902→
   903→    // Admin (via company WhatsApp)
   904→    try {
   905→      const r = await sendWhatsApp({
   906→        to: ADMIN_PHONE,
   907→        body: buildMsg('Admin', 'admin'),
   908→      })
   909→      if (r.success) sent++
   910→    } catch {}
   911→
   912→    return sent
   913→  }
